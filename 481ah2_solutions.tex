\textbf{Problem 1.}

Aitken's $\Delta^2$ Method:
\[
\hat{p}_n = \frac{p_{n+2}\,p_n - p_{n+1}^2}{p_{n+2} - 2p_{n+1} + p_n}
\]

\textbf{(a)} Substitute $p_n = x$, $p_{n+1} = g(x)$, $p_{n+2} = g(g(x))$:
\[
\boxed{G(x) = \frac{g(g(x))\cdot x - [g(x)]^2}{g(g(x)) - 2g(x) + x}}
\]

\textbf{(b)} $g(x) = x + x^2$, fixed point $p = 0$. Then $g'(x) = 1 + 2x$, so $g'(0) = 1$.

$\forall\, p_0 > 0$: $g(p_0) = p_0 + p_0^2 > p_0$, so the iterates are strictly increasing. Since $|g'(0)| = 1$, the fixed point is not attracting.

\[
\boxed{\text{Diverges}}
\]

\textbf{(c)} Compute $g(g(x))$ with $g(x) = x + x^2$:
\begin{align*}
g(g(x)) &= g(x + x^2) = (x + x^2) + (x + x^2)^2 \\
         &= x + x^2 + x^2 + 2x^3 + x^4 = x + 2x^2 + 2x^3 + x^4
\end{align*}

Numerator of $G(x)$:
\begin{align*}
g(g(x))\cdot x - [g(x)]^2 &= x(x + 2x^2 + 2x^3 + x^4) - (x + x^2)^2 \\
&= (x^2 + 2x^3 + 2x^4 + x^5) - (x^2 + 2x^3 + x^4) = x^4 + x^5 = x^4(1 + x)
\end{align*}

Denominator of $G(x)$:
\begin{align*}
g(g(x)) - 2g(x) + x &= (x + 2x^2 + 2x^3 + x^4) - 2(x + x^2) + x = 2x^3 + x^4 = x^3(2 + x)
\end{align*}

\[
\boxed{G(x) = \frac{x^4(1+x)}{x^3(2+x)} = \frac{x(1+x)}{2+x}}
\]

\textbf{(d)} Using the quotient rule on $G(x) = \dfrac{x + x^2}{x + 2}$:
\[
G'(x) = \frac{(1 + 2x)(x + 2) - (x + x^2)(1)}{(x + 2)^2}
= \frac{x^2 + 4x + 2}{(x + 2)^2}
\]

\[
G'(0) = \frac{0 + 0 + 2}{(0 + 2)^2} = \frac{2}{4} = \frac{1}{2}
\]

Since $|G'(0)| = \dfrac{1}{2} < 1$, the fixed-point iteration $x = G(x)$ converges to $p = 0$.

\[
\boxed{\text{Converges (linear, rate } \tfrac{1}{2}\text{)}}
\]

\textbf{Problem 4.}

\textbf{(a)} $g(x) = \left(\dfrac{7}{x}\right)^{1/4}$:
\[
g'(x) = -\frac{1}{4}\cdot\frac{(7/x)^{1/4}}{x}
\implies g'(p^*) = -\frac{1}{4}\cdot\frac{p^*}{p^*} = -\frac{1}{4}
\implies |g'(p^*)| = 0.25
\]

\textbf{(b)} $g(x) = x - \dfrac{x^5 - 7}{x^2}$:
\[
g'(x) = 1 - \frac{5x^4 \cdot x^2 - (x^5 - 7)\cdot 2x}{x^4} = 1 - 3x^2 - \frac{14}{x^3}

\textbf{Problem 6.}

Let $g(x) = \dfrac{x}{2} + \dfrac{1}{x}$ on $[1, 2]$.

\textbf{Condition 1:} $g$ maps $[1, 2]$ into $[1, 2]$.

$g'(x) = \dfrac{1}{2} - \dfrac{1}{x^2}$. Setting $g'(x) = 0$: $x = \sqrt{2}$.

Since $g''(x) = 2/x^3 > 0$, the minimum of $g$ on $[1,2]$ is at $x = \sqrt{2}$:
\[
g(\sqrt{2}) = \frac{\sqrt{2}}{2} + \frac{1}{\sqrt{2}} = \frac{\sqrt{2}}{2} + \frac{\sqrt{2}}{2} = \sqrt{2} \approx 1.414
\]

At the endpoints:
\[
g(1) = \frac{1}{2} + 1 = \frac{3}{2} = 1.5, \qquad
g(2) = 1 + \frac{1}{2} = \frac{3}{2} = 1.5
\]

$g([1,2]) = [\sqrt{2},\; 3/2] \subset [1, 2]$. \checkmark

\textbf{Condition 2:} $|g'(x)| < 1$ on $(1, 2)$.
\[
g'(x) = \frac{1}{2} - \frac{1}{x^2} = \frac{x^2 - 2}{2x^2}
\]

$\forall\, x \in [1, 2]$: $|x^2 - 2| \le \max(|1 - 2|, |4 - 2|) = 2$ and $2x^2 \ge 2$, so:
\[
|g'(x)| = \frac{|x^2 - 2|}{2x^2} \le \frac{2}{2 \cdot 1} = 1
\]

For inequality on $(1, 2)$: $|x^2 - 2| < 2$ when $x \in (1, 2)$ and $2x^2 > 2$ when $x > 1$, so $|g'(x)| < 1$ on $(1,2)$. More precisely:
\[
\max_{x \in [1,2]} |g'(x)| = |g'(1)| = \left|\frac{1-2}{2}\right| = \frac{1}{2} < 1 \quad \checkmark
\]

\textbf{Fixed point:} $g(p) = p \implies \dfrac{p}{2} + \dfrac{1}{p} = p \implies \dfrac{1}{p} = \dfrac{p}{2} \implies p^2 = 2 \implies p = \sqrt{2}$.

$\{x_n\}$ converges to the unique fixed point $p = \sqrt{2}$ for any $x_0 \in [1, 2]$.

\[
\boxed{x_n \to \sqrt{2} \text{ for all } x_0 \in [1, 2]}
\]
